package javaapplication.pkg3.pkg6.resuelto;



/**
 *
 * @author JAVA
 */
public class JavaApplication36RESUELTO {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        int suma=0;
        for (int i = 1; i  <= 10;i ++){
            int impar = (i * 2) - 1;
            suma = suma+(impar);
    }
        
 
        System.out.println(suma);

    }
